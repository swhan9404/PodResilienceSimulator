**Create `src/visualization/MetricsChartManager.ts`:**

This class receives MetricsSample[] from the simulation snapshot and transforms them into uPlot-compatible AlignedData arrays with a 60-second sliding window.

```typescript
import type { MetricsSample } from '../simulation/types';

export type ChartId = 'workerUsage' | 'readyPods' | 'rate503' | 'responseTime';

// uPlot AlignedData = [xValues: number[], ...yValues: (number | null)[]]
export type AlignedData = [number[], ...(number | null)[][]];

export class MetricsChartManager {
  private windowSeconds: number = 60; // D-06: fixed 60s window
  private profileNames: string[] = [];
  private lastProcessedIndex: number = 0;

  constructor(profileNames: string[]) {
    this.profileNames = profileNames;
  }

  /**
   * Process new metrics samples and return chart data for all 4 charts.
   * Called at 1Hz (per D-10).
   * 
   * @param allSamples - Complete MetricsSample[] from snapshot.metrics
   * @param currentClockMs - Current simulation clock in ms
   */
  update(allSamples: MetricsSample[], currentClockMs: number): void {
    this.lastProcessedIndex = allSamples.length;
    // Data is derived on-demand in getChartData to avoid storing duplicates
  }

  getChartData(chartId: ChartId, allSamples: MetricsSample[], currentClockMs: number): AlignedData {
    // Apply 60-second sliding window
    const windowStartMs = Math.max(0, currentClockMs - this.windowSeconds * 1000);
    const windowed = allSamples.filter(s => s.time >= windowStartMs);

    const timeSeconds = windowed.map(s => s.time / 1000); // Pitfall 2: uPlot expects seconds

    switch (chartId) {
      case 'workerUsage': {
        // MET-01: (activeWorkerCount / totalWorkerCount) * 100
        const values = windowed.map(s =>
          s.totalWorkerCount > 0 ? (s.activeWorkerCount / s.totalWorkerCount) * 100 : 0
        );
        return [timeSeconds, values];
      }
      case 'readyPods': {
        // MET-02: readyPodCount directly
        const values = windowed.map(s => s.readyPodCount);
        return [timeSeconds, values];
      }
      case 'rate503': {
        // MET-03: cumulative 503 rate within the sample window
        // Each sample has per-second counts. Compute rate as:
        // (sample.total503s / max(sample.totalRequests + sample.total503s, 1)) * 100
        const values = windowed.map(s => {
          const total = s.totalRequests + s.total503s;
          return total > 0 ? (s.total503s / total) * 100 : 0;
        });
        return [timeSeconds, values];
      }
      case 'responseTime': {
        // MET-04: one series per profile, average response time in ms
        const series: (number | null)[][] = this.profileNames.map(name =>
          windowed.map(s => {
            const profile = s.perProfileResponseTime[name];
            if (profile && profile.count > 0) return profile.sum / profile.count;
            return null;
          })
        );
        return [timeSeconds, ...series];
      }
    }
  }

  getProfileNames(): string[] {
    return this.profileNames;
  }
}
```

Key implementation rules:
- Time values MUST be divided by 1000 (simulation uses ms, uPlot expects seconds) per Pitfall 2
- Sliding window: filter samples where `sample.time >= currentClockMs - 60000` per D-06
- Worker usage: `(activeWorkerCount / totalWorkerCount) * 100`, range 0-100
- 503 rate: per-sample `(total503s / max(totalRequests + total503s, 1)) * 100`, range 0-100
- Ready pods: `readyPodCount` directly
- Response time: `perProfileResponseTime[profileName].sum / perProfileResponseTime[profileName].count` in ms, null if no data for that profile in that sample
- Response time chart returns `[timeArray, ...profileSeries]` where profileSeries has one array per profile name
