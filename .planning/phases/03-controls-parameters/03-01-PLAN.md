  // Config state (PAR-01 through PAR-06)
  config: SimulationConfig;
  updateConfig: (partial: Partial<SimulationConfig>) => void;
  updateLivenessProbe: (partial: Partial<ProbeConfig>) => void;
  updateReadinessProbe: (partial: Partial<ProbeConfig>) => void;
  setRequestProfiles: (profiles: RequestProfile[]) => void;

  statusReadyPods: number;

    workerUsage: AlignedData;

  resume: () => void;
  reset: () => void;
  stopRequests: () => void;
  setSpeed: (speed: number) => void;
  onRendererReady: (renderer: PodRenderer | null) => void;
  onCanvasResize: (width: number, height: number) => void;
}

Use `create<SimulationStore>()((set, get) => ({...}))` — the curried Zustand v5 TypeScript pattern.

Key action implementations:

**`start()` action** — must replicate ALL responsibilities from useSimulation.ts:
  a. Normalize ratios with `normalizeRatios()` helper (D-03): sum all profile ratios, divide each by total
  b. Create `new SimulationEngine(normalizedConfig)`
  c. Create `new MetricsChartManager(normalizedConfig.requestProfiles.map(p => p.name))`
  d. Create `new SimulationLoop(engine, chartManager, { onChartUpdate })` where onChartUpdate calls `set({ chartData: data })` AND includes throttled status update (see below)
  e. Connect renderer: `if (get().rendererRef.current) loop.setPodRenderer(get().rendererRef.current)`
  f. Set canvas dims: `if (get().canvasDimsRef.current.width > 0) loop.setCanvasDimensions(...)`
  g. Detect dark mode: `const darkQuery = window.matchMedia('(prefers-color-scheme: dark)'); loop.setDarkMode(darkQuery.matches)`
  h. Store darkQuery listener ref for cleanup in reset()
  i. `set({ engineRef: engine, loopRef: loop, playback: 'running' })`
  j. `loop.start()`

**Throttled status update** — inside the onChartUpdate callback (which fires at ~1Hz chart throttle), also update status:
```typescript
const snapshot = engine.getSnapshot();
set({
  statusClock: snapshot.clock,
  status503: snapshot.stats.total503s,
  statusReadyPods: snapshot.stats.readyPodCount,
});
```
This piggybacks on the existing chart throttle (~1Hz) which is sufficient for status display. No separate 4Hz timer needed since the chart callback already fires regularly.

**`pause()` action**: `get().loopRef?.stop(); set({ playback: 'paused' })`

**`resume()` action**: `get().loopRef?.start(); set({ playback: 'running' })`

**`reset()` action** (per Pitfall 2 — must stop loop BEFORE clearing refs):
  a. `get().loopRef?.stop()` — cancels rAF
  b. Remove dark mode listener if stored
  c. `set({ engineRef: null, loopRef: null, playback: 'idle', speed: 1, statusClock: 0, status503: 0, statusReadyPods: 0, chartData: EMPTY_CHART_DATA })`

**`stopRequests()` action**: `get().engineRef?.stopRequests(); set({ playback: 'stopped_requests' })`

**`setSpeed(speed)` action**: `get().loopRef?.setSpeed(speed); set({ speed })`

**`onRendererReady(renderer)` action**: `get().rendererRef.current = renderer; get().loopRef?.setPodRenderer(renderer)`

**`onCanvasResize(width, height)` action**: `get().canvasDimsRef.current = { width, height }; get().loopRef?.setCanvasDimensions(width, height)`

The `rendererRef` and `canvasDimsRef` should be created via `createRef()` style objects outside the store, initialized in the store's initial state as `{ current: null }` and `{ current: { width: 0, height: 0 } }` respectively.
