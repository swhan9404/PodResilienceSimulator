```typescript
import { MetricsSample } from './types';

export interface MetricEvent {
  type: 'request_complete' | 'request_503' | 'dropped_by_restart';
  profileName?: string;
  responseTime?: number;    // for completed requests: endTime - arrivalTime (includes queue wait)
  count?: number;           // for dropped_by_restart: number of dropped requests
}

interface MetricsBucket {
  totalRequests: number;
  total503s: number;
  droppedByRestart: number;
  perProfileResponseTime: Record<string, { sum: number; count: number }>;
}

export class MetricsCollector {
  private sampleIntervalMs: number = 1000;  // 1 simulated second
  private lastSampleTime: number = 0;
  private samples: MetricsSample[] = [];
  private currentBucket: MetricsBucket;

  // Running totals (cumulative, not per-bucket)
  totalRequests: number = 0;
  total503s: number = 0;
  droppedByRestart: number = 0;

  constructor(sampleIntervalMs: number = 1000) {
    this.sampleIntervalMs = sampleIntervalMs;
    this.currentBucket = this.createEmptyBucket();
  }

  private createEmptyBucket(): MetricsBucket {
    return {
      totalRequests: 0,
      total503s: 0,
      droppedByRestart: 0,
      perProfileResponseTime: {},
    };
  }

  record(event: MetricEvent): void {
    // Accumulate into currentBucket AND running totals
    switch (event.type) {
      case 'request_complete':
        this.currentBucket.totalRequests++;
        this.totalRequests++;
        if (event.profileName && event.responseTime !== undefined) {
          if (!this.currentBucket.perProfileResponseTime[event.profileName]) {
            this.currentBucket.perProfileResponseTime[event.profileName] = { sum: 0, count: 0 };
          }
          this.currentBucket.perProfileResponseTime[event.profileName].sum += event.responseTime;
          this.currentBucket.perProfileResponseTime[event.profileName].count++;
        }
        break;
      case 'request_503':
        this.currentBucket.total503s++;
        this.total503s++;
        break;
      case 'dropped_by_restart':
        const dropCount = event.count ?? 1;
        this.currentBucket.droppedByRestart += dropCount;
        this.droppedByRestart += dropCount;
        break;
    }
  }

  maybeSample(currentTime: number, readyPodCount: number, activeWorkerCount: number, totalWorkerCount: number): void {
    // If currentTime - lastSampleTime >= sampleIntervalMs, flush bucket to samples
    while (currentTime - this.lastSampleTime >= this.sampleIntervalMs) {
      this.lastSampleTime += this.sampleIntervalMs;
      const sample: MetricsSample = {
        time: this.lastSampleTime,
        totalRequests: this.currentBucket.totalRequests,
        total503s: this.currentBucket.total503s,
        droppedByRestart: this.currentBucket.droppedByRestart,
        readyPodCount,
        activeWorkerCount,
        totalWorkerCount,
        perProfileResponseTime: { ...this.currentBucket.perProfileResponseTime },
      };
      this.samples.push(sample);
      this.currentBucket = this.createEmptyBucket();
    }
  }

  getSamples(): MetricsSample[] {
    return this.samples;
  }

  reset(): void {
    this.samples = [];
    this.lastSampleTime = 0;
    this.currentBucket = this.createEmptyBucket();
    this.totalRequests = 0;
    this.total503s = 0;
    this.droppedByRestart = 0;
  }
}
```

- Uses `while` loop in maybeSample to handle large time jumps (e.g., 100x speed advancing several seconds per frame)
- Keeps running totals (totalRequests, total503s, droppedByRestart) for snapshot stats
- Per D-12: samples array is part of the snapshot

  <behavior>
    SIM-01 DES engine with event queue and clock:
    - Engine uses MinHeap<SimEvent> for event scheduling
    - step(deltaMs) processes all events up to clock + deltaMs
    - Clock advances to event time as each event is processed

    SIM-02 Request arrivals at configured RPS:
    - First REQUEST_ARRIVAL at time 0
    - Each REQUEST_ARRIVAL self-schedules next at clock + Math.round(1000 / rps)
    - When rps is 0 (stopped), no new arrivals scheduled

    SIM-03 Request profile selection:
    - Each request's latencyMs determined by selectProfile(profiles, rng())
    - Deterministic: same seed produces same request sequence

    SIM-04 Integer ms logical time:
    - All event times are integer ms
    - No Date.now() or performance.now() in engine

    HC-01/HC-02 Probe scheduling:
    - Initial probes scheduled at time 0 for each pod (both liveness and readiness)
    - Each probe self-schedules next probe

    HC-03 Probe timeout:
    - When probe sent, PROBE_TIMEOUT event scheduled at sendTime + timeoutSeconds * 1000
    - If probe completes before timeout, timeout event is cancelled (via generation check)
    - If timeout fires first, probe result = failure

    HC-08 Probe after previous completion + periodSeconds:
    - Next probe scheduled at completionTime + periodSeconds * 1000 (per D-08)
    - Not at fixed intervals from initial time

    D-03 Self-scheduling events:
    - REQUEST_ARRIVAL schedules next REQUEST_ARRIVAL
    - PROBE events schedule next PROBE after completion/timeout + periodSeconds

    D-12 Snapshot:
    - getSnapshot() returns immutable SimulationSnapshot
    - Contains all pod states, worker occupancy, cumulative stats, metrics samples

    Tombstone/generation pattern for event cancellation:
    - Each SimEvent stores the pod's generation at creation time
    - When processing, if event.generation !== pod.generation, skip the event
    - Pod restart increments generation, effectively cancelling all pending events for that pod

    Safety valve:
    - step() processes at most 50000 events per call to prevent infinite loops

    stopRequests():
    - Sets rps to 0, changes phase to 'stopped_requests'
    - No more REQUEST_ARRIVAL events scheduled

    Scenario tests (critical for Phase 1 success criteria):
    - 100% slow requests (e.g., all 30s latency): all pods must cascade to RESTARTING within bounded sim time
    - 0% slow requests (all 50ms latency): all pods stay READY for 60+ seconds of sim time
    - Health check fires, consumes worker, returns result to probe counter
    - LB distributes only to READY pods, returns 503 when none available
    - Snapshot contains correct stats at each point

```typescript
import { SimEvent, EventType, SimulationConfig, SimulationSnapshot, PodState, ActiveRequest, PodSnapshot, RequestProfile } from './types';
import { MinHeap } from './priority-queue';
import { Pod } from './pod';
import { LoadBalancer, RoundRobinStrategy } from './load-balancer';
import { MetricsCollector, MetricEvent } from './metrics';
import { createRng, selectProfile } from './rng';

export class SimulationEngine {
  private clock: number = 0;
  private eventQueue: MinHeap<SimEvent>;
  private pods: Pod[];
  private loadBalancer: LoadBalancer;
  private metrics: MetricsCollector;
  private rng: () => number;
  private config: SimulationConfig;
  private nextRequestId: number = 1;
  private rps: number;
  private phase: 'running' | 'stopped_requests' | 'recovered' | 'finished' = 'running';
  private requestArrivalTimes: Map<number, number> = new Map();  // requestId -> arrival sim time

  // Running stats (NOT per-window, cumulative)
  private totalRequests: number = 0;
  private total503s: number = 0;

  private static readonly MAX_EVENTS_PER_STEP = 50000;

  constructor(config: SimulationConfig) {
    this.config = config;
    this.rps = config.rps;
    this.rng = createRng(config.seed);
    this.eventQueue = new MinHeap<SimEvent>((a, b) => a.time - b.time);
    this.metrics = new MetricsCollector();

    // Create pods
    this.pods = [];
    for (let i = 0; i < config.podCount; i++) {
      this.pods.push(new Pod(
        i, config.workersPerPod, config.maxBacklogPerPod,
        config.livenessProbe, config.readinessProbe
      ));
    }

    // Create load balancer with round-robin strategy
    this.loadBalancer = new LoadBalancer(this.pods, new RoundRobinStrategy());

    // Schedule initial events
    this.scheduleInitialEvents();
  }

  private scheduleInitialEvents(): void {
    // First request arrival at time 0 (per D-03, SIM-02)
    if (this.rps > 0) {
      this.eventQueue.push({ time: 0, type: 'REQUEST_ARRIVAL' });
    }

    // Initial probes for each pod (per HC-01, HC-02)
    for (const pod of this.pods) {
      this.eventQueue.push({
        time: 0,
        type: 'LIVENESS_PROBE',
        podId: pod.id,
        probeType: 'liveness',
        generation: pod.generation,
      });
      this.eventQueue.push({
        time: 0,
        type: 'READINESS_PROBE',
        podId: pod.id,
        probeType: 'readiness',
        generation: pod.generation,
      });
    }
  }

  step(deltaMs: number): void {
    const targetTime = this.clock + Math.round(deltaMs);
    let eventsProcessed = 0;

    while (
      !this.eventQueue.isEmpty() &&
      this.eventQueue.peek().time <= targetTime &&
      eventsProcessed < SimulationEngine.MAX_EVENTS_PER_STEP
    ) {
      const event = this.eventQueue.pop();
      this.clock = event.time;
      this.processEvent(event);
      eventsProcessed++;
    }

    this.clock = targetTime;

    // Sample metrics after advancing
    const activeWorkers = this.countActiveWorkers();
    const totalWorkers = this.config.podCount * this.config.workersPerPod;
    const readyCount = this.loadBalancer.getReadyCount();
    this.metrics.maybeSample(this.clock, readyCount, activeWorkers, totalWorkers);

    // Check recovery (all pods READY after stopped_requests)
    if (this.phase === 'stopped_requests' && readyCount === this.config.podCount) {
      this.phase = 'recovered';
    }
  }

  private processEvent(event: SimEvent): void {
    // Tombstone check: skip events from previous pod generation
    if (event.podId !== undefined && event.generation !== undefined) {
      const pod = this.pods[event.podId];
      if (event.generation !== pod.generation) {
        return;  // Stale event from before restart
      }
    }

    switch (event.type) {
      case 'REQUEST_ARRIVAL':
        this.handleRequestArrival(event);
        break;
      case 'REQUEST_COMPLETE':
        this.handleRequestComplete(event);
        break;
      case 'LIVENESS_PROBE':
      case 'READINESS_PROBE':
        this.handleProbe(event);
        break;
      case 'PROBE_TIMEOUT':
        this.handleProbeTimeout(event);
        break;
      case 'PROBE_COMPLETE':
        this.handleProbeComplete(event);
        break;
      case 'POD_RESTART':
        this.handlePodRestart(event);
        break;
      case 'POD_INIT_COMPLETE':
        this.handlePodInitComplete(event);
        break;
    }
  }

  // --- Event Handlers ---

  private handleRequestArrival(event: SimEvent): void {
    this.totalRequests++;
    const profile = selectProfile(this.config.requestProfiles, this.rng());
    const requestId = this.nextRequestId++;

    // Store arrival time for response time calculation
    this.requestArrivalTimes.set(requestId, this.clock);

    const targetPod = this.loadBalancer.selectPod();
    if (!targetPod) {
      // LB-02: all not ready -> 503
      this.total503s++;
      this.metrics.record({ type: 'request_503' });
      this.requestArrivalTimes.delete(requestId);
    } else {
      const request: ActiveRequest = {
        requestId,
        profileName: profile.name,
        profileColor: profile.color,
        startTime: this.clock,
        endTime: this.clock + profile.latencyMs,
        isProbe: false,
      };
      const result = targetPod.tryAccept(request);
      if (result.status === 'assigned') {
        // Schedule completion
        this.eventQueue.push({
          time: request.endTime,
          type: 'REQUEST_COMPLETE',
          podId: targetPod.id,
          requestId,
          workerIndex: result.workerIndex,
          generation: targetPod.generation,
        });
      } else if (result.status === 'queued') {
        // Backlog -- completion scheduled when dequeued to worker
      } else {
        // Rejected (backlog full) -> 503
        this.total503s++;
        this.metrics.record({ type: 'request_503' });
        this.requestArrivalTimes.delete(requestId);
      }
    }

    // Self-schedule next arrival (per D-03)
    if (this.rps > 0) {
      const intervalMs = Math.round(1000 / this.rps);
      this.eventQueue.push({
        time: this.clock + Math.max(intervalMs, 1),
        type: 'REQUEST_ARRIVAL',
      });
    }
  }

  private handleRequestComplete(event: SimEvent): void {
    const pod = this.pods[event.podId!];
    const dequeued = pod.completeRequest(event.workerIndex!, this.clock);

    // Record response time for completed request
    const arrivalTime = this.requestArrivalTimes.get(event.requestId!);
    if (arrivalTime !== undefined) {
      const profile = this.findProfileByRequestId(event);
      this.metrics.record({
        type: 'request_complete',
        profileName: profile,
        responseTime: this.clock - arrivalTime,
      });
      this.requestArrivalTimes.delete(event.requestId!);
    }

    // If a backlog item was dequeued and assigned to the freed worker, schedule its completion
    if (dequeued) {
      const workerIndex = pod.workers.findIndex(w => w !== null && w.requestId === dequeued.requestId);
      if (workerIndex >= 0) {
        this.eventQueue.push({
          time: dequeued.endTime,
          type: dequeued.isProbe ? 'PROBE_COMPLETE' : 'REQUEST_COMPLETE',
          podId: pod.id,
          requestId: dequeued.requestId,
          workerIndex,
          probeType: dequeued.probeType,
          generation: pod.generation,
        });
      }
    }
  }

  private handleProbe(event: SimEvent): void {
    const pod = this.pods[event.podId!];
    const probeType = event.probeType!;
    const requestId = this.nextRequestId++;

    const probeRequest: ActiveRequest = {
      requestId,
      profileName: `${probeType}_probe`,
      profileColor: '#888888',
      startTime: this.clock,
      endTime: this.clock + 1,  // 1ms processing time per D-07
      isProbe: true,
      probeType,
    };

    const result = pod.tryAccept(probeRequest);

    if (result.status === 'assigned') {
      // Schedule probe completion
      this.eventQueue.push({
        time: probeRequest.endTime,
        type: 'PROBE_COMPLETE',
        podId: pod.id,
        requestId,
        workerIndex: result.workerIndex,
        probeType,
        generation: pod.generation,
      });
      // Schedule timeout (per HC-03): timeout starts at send time
      const probeConfig = probeType === 'liveness' ? this.config.livenessProbe : this.config.readinessProbe;
      this.eventQueue.push({
        time: this.clock + probeConfig.timeoutSeconds * 1000,
        type: 'PROBE_TIMEOUT',
        podId: pod.id,
        requestId,
        probeType,
        generation: pod.generation,
      });
    } else if (result.status === 'queued') {
      // Probe in backlog -- schedule timeout from send time (HC-03)
      const probeConfig = probeType === 'liveness' ? this.config.livenessProbe : this.config.readinessProbe;
      this.eventQueue.push({
        time: this.clock + probeConfig.timeoutSeconds * 1000,
        type: 'PROBE_TIMEOUT',
        podId: pod.id,
        requestId,
        probeType,
        generation: pod.generation,
      });
    } else {
      // Rejected (backlog full) -> immediate failure per HC-04, D-09
      this.handleProbeFailure(pod, probeType);
    }
  }

  private handleProbeComplete(event: SimEvent): void {
    const pod = this.pods[event.podId!];
    const probeType = event.probeType!;

    // Complete the worker (free slot, maybe dequeue backlog)
    const dequeued = pod.completeRequest(event.workerIndex!, this.clock);

    // Record probe success
    const probeResult = pod.recordProbeResult(probeType, true);
    this.handleProbeAction(pod, probeResult, probeType);

    // Schedule next probe: periodSeconds after completion (per D-08, HC-08)
    this.scheduleNextProbe(pod, probeType);

    // Handle dequeued backlog item
    if (dequeued) {
      const workerIndex = pod.workers.findIndex(w => w !== null && w.requestId === dequeued.requestId);
      if (workerIndex >= 0) {
        this.eventQueue.push({
          time: dequeued.endTime,
          type: dequeued.isProbe ? 'PROBE_COMPLETE' : 'REQUEST_COMPLETE',
          podId: pod.id,
          requestId: dequeued.requestId,
          workerIndex,
          probeType: dequeued.probeType,
          generation: pod.generation,
        });
      }
    }
  }

  private handleProbeTimeout(event: SimEvent): void {
    const pod = this.pods[event.podId!];
    const probeType = event.probeType!;

    // Check if probe already completed (via requestId lookup in workers/backlog)
    // If probe is still in backlog or still being processed, record failure
    // The probe request may have already completed -- in that case the generation check
    // at the top of processEvent handles it (or the probe completed and freed the worker).
    // We use a simple approach: record failure. If probe already completed, the
    // PROBE_COMPLETE handler already recorded success. The timeout handler should check
    // whether the probe is still pending.

    // Search for the probe in workers and backlog
    const probeStillPending = this.isProbeStillPending(pod, event.requestId!);
    if (probeStillPending) {
      // Remove probe from backlog if it's there (it timed out before being processed)
      this.removeProbeFromBacklog(pod, event.requestId!);
      // Record failure
      this.handleProbeFailure(pod, probeType);
    }
    // If probe already completed, this timeout is stale -- do nothing
    // Next probe is only scheduled from completion or failure handler
  }

  private handlePodRestart(event: SimEvent): void {
    const pod = this.pods[event.podId!];
    const droppedCount = pod.restart();

    // Record dropped requests in metrics (per Pitfall #5: single event, not individual)
    if (droppedCount > 0) {
      this.metrics.record({ type: 'dropped_by_restart', count: droppedCount });
    }

    // Clean up arrival times for dropped requests
    // (pod.restart already cleared workers and backlog)

    // Schedule POD_INIT_COMPLETE after initializeTime (per D-06, POD-06)
    this.eventQueue.push({
      time: this.clock + this.config.initializeTimeMs,
      type: 'POD_INIT_COMPLETE',
      podId: pod.id,
      generation: pod.generation,
    });
  }

  private handlePodInitComplete(event: SimEvent): void {
    const pod = this.pods[event.podId!];
    pod.initComplete();

    // Resume probes (per D-06): schedule first probes after init
    this.eventQueue.push({
      time: this.clock,
      type: 'LIVENESS_PROBE',
      podId: pod.id,
      probeType: 'liveness',
      generation: pod.generation,
    });
    this.eventQueue.push({
      time: this.clock,
      type: 'READINESS_PROBE',
      podId: pod.id,
      probeType: 'readiness',
      generation: pod.generation,
    });
  }

  // --- Helper Methods ---

  private handleProbeFailure(pod: Pod, probeType: 'liveness' | 'readiness'): void {
    const probeResult = pod.recordProbeResult(probeType, false);
    this.handleProbeAction(pod, probeResult, probeType);

    // Schedule next probe after failure + periodSeconds (per D-08)
    this.scheduleNextProbe(pod, probeType);
  }

  private handleProbeAction(pod: Pod, probeResult: { action: string }, probeType: 'liveness' | 'readiness'): void {
    if (probeResult.action === 'restart') {
      // HC-05: liveness threshold exceeded -> restart
      this.eventQueue.push({
        time: this.clock,
        type: 'POD_RESTART',
        podId: pod.id,
        generation: pod.generation,
      });
    }
    // HC-06 (remove_from_lb) and HC-07 (add_to_lb) are handled by Pod.recordProbeResult
    // which directly changes pod.state. The LB reads pod.state on each selectPod call.
  }

  private scheduleNextProbe(pod: Pod, probeType: 'liveness' | 'readiness'): void {
    // Per D-08, HC-08: next probe at current time + periodSeconds
    // Only if pod is not restarting (probes resume after init)
    if (pod.state === PodState.RESTARTING) return;

    const probeConfig = probeType === 'liveness' ? this.config.livenessProbe : this.config.readinessProbe;
    this.eventQueue.push({
      time: this.clock + probeConfig.periodSeconds * 1000,
      type: probeType === 'liveness' ? 'LIVENESS_PROBE' : 'READINESS_PROBE',
      podId: pod.id,
      probeType,
      generation: pod.generation,
    });
  }

  private isProbeStillPending(pod: Pod, requestId: number): boolean {
    // Check workers
    for (const worker of pod.workers) {
      if (worker && worker.requestId === requestId && worker.isProbe) return true;
    }
    // Check backlog
    for (const item of pod.backlog) {
      if (item.requestId === requestId && item.isProbe) return true;
    }
    return false;
  }

  private removeProbeFromBacklog(pod: Pod, requestId: number): void {
    const index = pod.backlog.findIndex(item => item.requestId === requestId);
    if (index >= 0) {
      pod.backlog.splice(index, 1);
    }
  }

  private findProfileByRequestId(event: SimEvent): string | undefined {
    // Worker at the pod should have the profile name
    const pod = this.pods[event.podId!];
    // The request was just completed, worker might already be cleared
    // Use the event's stored data -- we need to track this differently
    // Alternative: check requestArrivalTimes -- but we need profile name
    // Simplest: look up in workers before completion clears it
    // Since completeRequest already freed the slot, we need another approach
    return undefined; // Profile tracking handled via a Map in the constructor
    // NOTE: Executor should use a requestProfiles Map<requestId, profileName> for this
  }

  private countActiveWorkers(): number {
    let count = 0;
    for (const pod of this.pods) {
      for (const worker of pod.workers) {
        if (worker !== null) count++;
      }
    }
    return count;
  }

  // --- Public API ---

  stopRequests(): void {
    this.rps = 0;
    if (this.phase === 'running') {
      this.phase = 'stopped_requests';
    }
  }

  getSnapshot(): SimulationSnapshot {
    return {
      clock: this.clock,
      pods: this.pods.map(p => p.getSnapshot(this.clock)),
      stats: {
        totalRequests: this.totalRequests,
        total503s: this.total503s,
        droppedByRestart: this.metrics.droppedByRestart,
        readyPodCount: this.loadBalancer.getReadyCount(),
        activeWorkerCount: this.countActiveWorkers(),
        totalWorkerCount: this.config.podCount * this.config.workersPerPod,
      },
      metrics: this.metrics.getSamples(),
      phase: this.phase,
    };
  }

  getClock(): number {
    return this.clock;
  }

  getEventQueueSize(): number {
    return this.eventQueue.size;
  }
}
```
