  <behavior>
    POD-01 State machine:
    - New Pod starts in READY state
    - Pod can transition READY -> NOT_READY (readiness failureThreshold)
    - Pod can transition NOT_READY -> READY (readiness successThreshold)
    - Pod can transition READY -> RESTARTING (liveness failureThreshold)
    - Pod can transition NOT_READY -> RESTARTING (liveness failureThreshold)
    - Pod can transition RESTARTING -> NOT_READY (after initComplete called)

    POD-02 Workers:
    - Pod with workersPerPod=4 has 4 worker slots
    - Idle worker accepts request, returns { status: 'assigned', workerIndex }
    - Busy worker slot is not available for new requests

    POD-03 Backlog:
    - When all workers busy, request goes to backlog, returns { status: 'queued' }
    - When worker completes, next backlog item is dequeued (FIFO order) and assigned

    POD-04 Backlog full:
    - When backlog at maxBacklogPerPod, new request returns { status: 'rejected', reason: 'backlog_full' }

    POD-05 Restart drops all:
    - Pod.restart() clears all workers (sets to null), clears backlog, returns count of dropped requests
    - Pod state becomes RESTARTING

    POD-06 Init complete:
    - Pod.initComplete() transitions RESTARTING -> NOT_READY
    - Probe counters reset to zero

    HC-01/HC-02 Probes use worker:
    - Probes use tryAccept same as regular requests (isProbe=true on ActiveRequest)
    - Probe processing time is 1ms (set by caller via endTime = startTime + 1)

    HC-03 Probe timeout:
    - No special Pod logic -- timeout tracking is engine's responsibility via PROBE_TIMEOUT events

    HC-04 Backlog full -> immediate probe failure:
    - When backlog full and tryAccept called with a probe request, returns { status: 'rejected', reason: 'backlog_full' }
    - (Same as regular request rejection -- engine interprets this as immediate probe failure)

    HC-05 Liveness failures -> restart:
    - Pod.recordProbeResult('liveness', false) increments consecutiveFailures, resets consecutiveSuccesses
    - When consecutiveFailures >= failureThreshold, returns { action: 'restart' }

    HC-06 Readiness failures -> not ready:
    - Pod.recordProbeResult('readiness', false) increments consecutiveFailures
    - When consecutiveFailures >= failureThreshold and pod is READY, transitions to NOT_READY, returns { action: 'remove_from_lb' }

    HC-07 Consecutive successes -> ready:
    - Pod.recordProbeResult('readiness', true) increments consecutiveSuccesses, resets consecutiveFailures
    - When consecutiveSuccesses >= successThreshold and pod is NOT_READY, transitions to READY, returns { action: 'add_to_lb' }

    HC-08 Probe scheduling:
    - No special Pod logic -- scheduling is engine's responsibility

    Pitfall #9 (NOT_READY vs RESTARTING):
    - NOT_READY: existing workers CONTINUE processing, backlog drains normally. Only new LB requests are blocked.
    - RESTARTING: ALL workers cleared, ALL backlog dropped.

    Pitfall #15 (FIFO backlog):
    - Backlog must be FIFO (Array.push + shift, or custom ring buffer). NOT LIFO.

    Snapshot:
    - Pod.getSnapshot(currentTime) returns PodSnapshot with worker progress calculated as (currentTime - startTime) / (endTime - startTime)
  </behavior>

```typescript
import { PodState, ActiveRequest, AcceptResult, ProbeCounter, ProbeConfig, WorkerSnapshot, PodSnapshot } from './types';

export interface ProbeResult {
  action: 'none' | 'remove_from_lb' | 'add_to_lb' | 'restart';
}

export class Pod {
  readonly id: number;
  state: PodState = PodState.READY;
  workers: (ActiveRequest | null)[];        // fixed-size array, null = idle
  backlog: ActiveRequest[] = [];            // FIFO queue (push to end, shift from front)
  readonly maxBacklog: number;
  livenessCounter: ProbeCounter = { consecutiveFailures: 0, consecutiveSuccesses: 0 };
  readinessCounter: ProbeCounter = { consecutiveFailures: 0, consecutiveSuccesses: 0 };
  livenessHistory: boolean[] = [];          // last N probe results
  readinessHistory: boolean[] = [];
  generation: number = 0;                   // incremented on restart for event cancellation
  private livenessConfig: ProbeConfig;
  private readinessConfig: ProbeConfig;
  private readonly historySize: number = 10;

  constructor(id: number, workersPerPod: number, maxBacklog: number, livenessConfig: ProbeConfig, readinessConfig: ProbeConfig) { ... }

  tryAccept(request: ActiveRequest): AcceptResult { ... }
  // 1. If state is RESTARTING -> reject with reason 'not_accepting'
  // 2. Find first null worker slot -> assign, return { status: 'assigned', workerIndex }
  // 3. If backlog.length < maxBacklog -> push to backlog (FIFO), return { status: 'queued' }
  // 4. Else -> reject with reason 'backlog_full'
  // NOTE: NOT_READY pods can still accept if called directly (engine controls whether LB sends to them)

  completeRequest(workerIndex: number, currentTime: number): ActiveRequest | null { ... }
  // 1. Set workers[workerIndex] = null
  // 2. If backlog non-empty -> shift first item, assign to freed worker, return the dequeued request
  // 3. Return null if backlog was empty

  recordProbeResult(probeType: 'liveness' | 'readiness', success: boolean): ProbeResult { ... }
  // Liveness:
  //   success -> reset failures to 0, increment successes
  //   failure -> reset successes to 0, increment failures
  //   if failures >= failureThreshold -> return { action: 'restart' }
  // Readiness:
  //   success -> reset failures to 0, increment successes
  //     if successes >= successThreshold AND state === NOT_READY -> state = READY, return { action: 'add_to_lb' }
  //   failure -> reset successes to 0, increment failures
  //     if failures >= failureThreshold AND state === READY -> state = NOT_READY, return { action: 'remove_from_lb' }
  // Push result to history (trim to historySize)
  // Default return: { action: 'none' }

  restart(): number { ... }
  // 1. Count busy workers + backlog length = dropped count
  // 2. Set all workers to null
  // 3. Clear backlog
  // 4. state = RESTARTING
  // 5. generation++
  // 6. Reset both probe counters to { consecutiveFailures: 0, consecutiveSuccesses: 0 }
  // 7. Clear probe histories
  // 8. Return dropped count

  initComplete(): void { ... }
  // 1. Assert state === RESTARTING
  // 2. state = NOT_READY
  // 3. Reset probe counters (fresh start)

  getSnapshot(currentTime: number): PodSnapshot { ... }
  // Build snapshot with worker progress = busy ? clamp((currentTime - startTime) / (endTime - startTime), 0, 1) : 0
}
```

```typescript
import { LoadBalancerStrategy } from './types';
import { Pod } from './pod';
import { PodState } from './types';

export class RoundRobinStrategy implements LoadBalancerStrategy {
  name = 'round-robin';
  private index: number = 0;

  selectPod(readyPodIds: number[]): number {
    // Return readyPodIds[this.index % readyPodIds.length], then increment index
    // readyPodIds is guaranteed non-empty when this is called
    const id = readyPodIds[this.index % readyPodIds.length];
    this.index++;
    return id;
  }

  reset(): void {
    this.index = 0;
  }
}

export class LoadBalancer {
  private strategy: LoadBalancerStrategy;
  private pods: Pod[];
  private lastReadySetKey: string = '';  // for detecting composition changes

  constructor(pods: Pod[], strategy: LoadBalancerStrategy) {
    this.pods = pods;
    this.strategy = strategy;
  }

  selectPod(): Pod | null {
    // 1. Filter pods where state === PodState.READY
    // 2. If none ready -> return null (per LB-02, D-11)
    // 3. Build sorted readyPodIds array
    // 4. Check if ready-set composition changed (compare ids as string key)
    //    If changed -> strategy.reset() (per Pitfall #7)
    //    Update lastReadySetKey
    // 5. Call strategy.selectPod(readyPodIds) to get pod id
    // 6. Return the Pod with that id
  }

  getReadyPods(): Pod[] {
    return this.pods.filter(p => p.state === PodState.READY);
  }

  getReadyCount(): number {
    return this.pods.filter(p => p.state === PodState.READY).length;
  }
}
```
