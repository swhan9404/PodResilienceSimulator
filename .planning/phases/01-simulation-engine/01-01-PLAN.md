       total503s: number;

   - Export `function createRng(seed: number): () => number` -- mulberry32 PRNG
     ```typescript
     export function createRng(seed: number): () => number {
       let s = seed | 0;
       return () => {
         s = (s + 0x6D2B79F5) | 0;
         let t = Math.imul(s ^ (s >>> 15), s | 1);
         t ^= t + Math.imul(t ^ (t >>> 7), t | 61);
         return ((t ^ (t >>> 14)) >>> 0) / 4294967296;
       };
     }
     ```
